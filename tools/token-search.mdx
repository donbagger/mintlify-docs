const resultsTitle = document.getElementById('results-title');
const resultsSubtitle = document.getElementById('results-subtitle');

// Search function - uses real API with fallback
async function searchTokens(query) {
  if (!query || query.trim().length < 1) return [];
  
  try {
    // Try to use the real API first
    if (window.tokenSearchAPI) {
      return await window.tokenSearchAPI.searchTokens(query);
    }
  } catch (error) {
    console.warn('API search failed, using fallback:', error);
  }
  
  // Fallback to local search
  return searchTokensLocal(query);
}

// Display results

  searchResults.style.display = 'block';
  loadingSpinner.style.display = 'none';
  
  if (tokens.length === 0) {
    resultsTableContainer.style.display = 'none';

    resultsSubtitle.textContent = `No tokens found matching "${query}"`;

  resultsTableContainer.style.display = 'block';
  noResults.style.display = 'none';
  resultsTitle.textContent = `Found ${tokens.length} Token${tokens.length > 1 ? 's' : ''}`;
  resultsSubtitle.textContent = `Results for "${query}"`;
  
  // Clear previous results
  resultsTableBody.innerHTML = '';
  
  // Add results to table
  tokens.forEach(token => {
    const row = document.createElement('tr');

// Show loading state
function showLoading() {
  searchResults.style.display = 'block';
  loadingSpinner.style.display = 'block';
  resultsTableContainer.style.display = 'none';
  noResults.style.display = 'none';
}

// Handle search
async function handleSearch() {
  const query = searchInput.value.trim();
  
  if (!query) {
    alert('Please enter a search term');
    return;
  }
  
  showLoading();
  
  try {
    const results = await searchTokens(query);
    displayResults(results, query);
  } catch (error) {
    console.error('Search error:', error);
    displayResults([], query);
  }
}

## How to Use

1. **Enter a token symbol or name** in the search box above
2. **Click "Search"** or press Enter
3. **Review the results** to see if your token is available
4. **Click "View API"** to see the API endpoint for that token
5. **Click "Details"** for more comprehensive token information

## What You'll See

- **Token Symbol & Name**: The official token identifier
- **Network**: Which blockchain the token is on (Ethereum, Solana, etc.)
- **Address**: The contract address for the token
- **Status**: Whether the token is available in our API
- **Actions**: Links to API documentation and detailed information

## For BD Teams

This tool helps you quickly verify token coverage before client discussions. You can:

- **Pre-sales calls**: Check if a client's tokens are supported
- **Partnership discussions**: Verify coverage for potential partners
- **Client onboarding**: Ensure all required tokens are available
- **Market research**: Explore what tokens are currently supported
